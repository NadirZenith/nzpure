# nzpure


# fast start

      - clone into wp-content/themes/
            git clone https://github.com/NadirZenith/nzpure

      - install node dps
            npm install

      - compile assets
            development - grunt dev
            production - grunt build
            
      - wp-config settings
            development - define( 'WP_ENV', 'development' );
            production - default
